# throttle

```javascript
/**
 * Функция для троттлинга вызова другой функции.
 * @param {Function} fn - Функция, которую нужно троттлировать.
 * @param {number} delay - Задержка в миллисекундах, через которую функция может быть вызвана снова.
 * @returns {Function} - Возвращаемая функция, которая выполняет троттлинг.
 */
const throttle = (fn, delay) => {
  let wait = false; // Флаг, указывающий, находится ли функция в состоянии ожидания.

  return function (...args) {
    if (wait) return; // Если функция находится в состоянии ожидания, она не выполняется.

    fn.apply(this, args); // Вызов функции с текущим контекстом и переданными аргументами.
    wait = true; // Установка флага ожидания в true.

    setTimeout(() => {
      wait = false; // Сброс флага ожидания через заданную задержку.
    }, delay);
  };
};
```
